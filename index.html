#/bin/bash
ndap_query=$1
sqlplus_account=$2
f_kapp_id=$3
f_bosang_cd=$4

#select 결과를 파일로 저장할 ndap, 웹서버 경로
data_file_name="always_memat_meta_data.dat"
#ndap_file_path=/data/riakshare/riak_data/ndapresult/${f_bosang_cd}/always_memat_meta/${data_file_name}

ndap_file_path=/data/riakshare/riak_data/ndapresult/${f_bosang_cd}/always_memat_meta
riakshare_file_path=/data/riakshare/riak_data/ndapresult/${f_bosang_cd}/always_memat_meta/

#디렉토리 생성
ssh ndap@222.122.131.175 mkdir -p $ndap_file_path
mkdir -p $riakshare_file_path

#조회결과파일 패스 설정
ndap_file_path=/data/riakshare/riak_data/ndapresult/${f_bosang_cd}/always_memat_meta/${data_file_name}

#메타코드 테이블 현재 max 값 구하기
max_meta_cd=$(
ssh ndap@222.122.131.175 << :QUERY
hive -e "select nvl(max(f_meta_cd),0) as max_meta_cd from kapp.cm_metacode_always where p_bosang_cd='"$f_bosang_cd"'" 
:QUERY)

#메타코드 테이블에 등록되지 않은 로그 테이블의 신규 데이터 insert
ssh ndap@222.122.131.175 << :QUERY
hive -e \
"insert into kapp.cm_metacode_always partition(p_bosang_cd='"$f_bosang_cd"')
select t1.f_album_sch_1, t1.f_song_sch_1, t1.f_artist_sch_1, concat(upper('"$f_bosang_cd"AM_') ,"$max_meta_cd"+row_number() over()) as f_meta_cd
from 
(select f_album_sch_1, f_song_sch_1, f_artist_sch_1 from kapp."$f_bosang_cd"_log group by f_album_sch_1, f_song_sch_1, f_artist_sch_1) t1
left join 
(select * from kapp.cm_metacode_always where p_bosang_cd='"$f_bosang_cd"') t2
on t1.f_album_sch_1 = t2.f_album_sch_1
and t1.f_song_sch_1=t2.f_song_sch_1
and t1.f_artist_sch_1=t2.f_artist_sch_1
where t2.f_album_sch_1 is null
and t2.f_song_sch_1 is null
and t2.f_artist_sch_1 is null"
:QUERY

#ndap 서버에 쿼리 실행결과 파일 생성.
ssh ndap@222.122.131.175 'rm -rf '${ndap_file_path}
ssh ndap@222.122.131.175 << :QUERY
hive -e \
"select * from (
select taa.f_bosang_cd, taa.f_album_nm, taa.f_song_nm, taa.f_artist_nm, taa.f_album_sch_1, taa.f_song_sch_1, taa.f_artist_sch_1, taa.f_count, taa.f_price, tb.f_meta_cd from 
(
    select 
    f_bosang_cd, max(f_album_nm) as f_album_nm, max(f_song_nm) as f_song_nm, max(f_artist_nm) as f_artist_nm, 
    f_album_sch_1, f_song_sch_1, f_artist_sch_1,   
    sum(f_count) as f_count, sum(f_price) as f_price  
    from  
    ( 
        select 
        t1.f_bosang_cd, 
        t1.f_album_nm, 
        t1.f_song_nm, 
        t1.f_artist_nm, 
        t1.f_count,
        t1.f_price,
        t1.f_album_sch_1,t1.f_song_sch_1,t1.f_artist_sch_1, t1.p_yyyyqq    
        from 
        (select f_log_no, f_yyyyqq, f_bosang_cd,f_album_nm,f_song_nm,f_artist_nm,f_count,f_price, f_album_sch_1,f_song_sch_1,f_artist_sch_1,p_yyyyqq from kapp."$f_bosang_cd"_log) t1
        left join 
        (select * from kapp."$f_bosang_cd"_log_matchinfo) t2
        on t1.p_yyyyqq=t2.f_yyyyqq and t1.f_log_no = t2.f_log_no 
        where t2.f_log_no is null
    ) ta group by  f_bosang_cd, f_album_sch_1, f_song_sch_1, f_artist_sch_1
) taa join kapp.cm_metacode_always tb on taa.f_bosang_cd=tb.p_bosang_cd and taa.f_album_sch_1=tb.f_album_sch_1 and taa.f_song_sch_1=tb.f_song_sch_1 and taa.f_artist_sch_1=tb.f_artist_sch_1
)taaa order by f_price desc" | sed 's/[\t]/\t/g'  > $ndap_file_path
:QUERY

#ndap 파일을 웹서버로 복사
scp ndap@222.122.131.175:$ndap_file_path $riakshare_file_path


#오라클로 임포트
#0.기 사용한 로더관련 파일 삭제
rm -rf ${riakshare_file_path}'/always_memat_meta.ctl'
rm -rf ${riakshare_file_path}'/always_memat_meta.bad'
rm -rf ${riakshare_file_path}'/always_memat_meta.dsc'
rm -rf ${riakshare_file_path}'/always_memat_meta.log'

#1.CONTROL 파일 생성

		F_PRICE \"REPLACE(REPLACE(REPLACE(TRIM(:F_PRICE),CHR(10),''),CHR(13),''),'NULL',0)\",

		F_KAPP_ID_REG \"'"${f_kapp_id}"'\",

		F_ALBUM_NM_KAPP FILLER,

		F_NAT_CD FILLER,
		F_MUSIC_DB_YYYYQQ FILLER,
		F_META_CD \":F_META_CD\"

exit

	sqlldr $sqlplus_account control=${riakshare_file_path}'always_memat_meta.ctl' \
